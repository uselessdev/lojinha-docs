---
title: "Recuperando Endereços"
api: "GET /v1/:customer/addresses"
description: "Recupera todos os endereços de um cliente"
---

<ParamField path="customer" type="string" required>
  `ID` ou `externalId` do cliente
</ParamField>

### Resposta

<ResponseField name="address" type="object">
  Endereço do cliente.

  <Expandable>
    <ResponseField name="id" type="uuid">
      ID do endereço
    </ResponseField>

    <ResponseField name="customerId" type="uuid">
      ID do cliente
    </ResponseField>

    <ResponseField name="alias" type="string">
      Nome para o endereço
    </ResponseField>

    <ResponseField name="zipcode" type="string">
      CEP do endereço
    </ResponseField>

    <ResponseField name="street" type="string">
      Nome da rua ou avenida
    </ResponseField>

    <ResponseField name="state" type="string">
      Estado
    </ResponseField>

    <ResponseField name="city" type="string">
      Cidade
    </ResponseField>

    <ResponseField name="country" type="string">
      País
    </ResponseField>

    <ResponseField name="neightborhood" type="string">
      Bairro
    </ResponseField>

    <ResponseField name="number" type="string">
      Número
    </ResponseField>

    <ResponseField name="complement" type="string">
      Complemento
    </ResponseField>

    <ResponseField name="createdAt" type="string">
      Data de criação
    </ResponseField>

    <ResponseField name="updatedAt" type="string">
      Data da última utilização
    </ResponseField>

    <ResponseField name="deletedAt" type="string">
      Data em que foi arquivado
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="number">
  Total de endereços que o cliente possui
</ResponseField>
